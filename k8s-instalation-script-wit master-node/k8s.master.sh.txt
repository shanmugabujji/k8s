#!/bin/bash
set -e

echo "===== Setting hostname ====="
hostnamectl set-hostname k8s-master

echo "===== Update /etc/hosts manually if needed ====="
# Example:
# 192.168.1.10   k8s-master

echo "===== Adding CentOS Stream repos ====="
cat <<EOF | tee /etc/yum.repos.d/base.repo
[baseos]
name=BaseOS
baseurl=http://mirror.stream.centos.org/9-stream/BaseOS/x86_64/os/
enabled=1
gpgcheck=0

[appstream]
name=AppStream
baseurl=http://mirror.stream.centos.org/9-stream/AppStream/x86_64/os/
enabled=1
gpgcheck=0
EOF

echo "===== Disabling SELinux ====="
setenforce 0 || true
sed -i 's/^SELINUX=.*/SELINUX=permissive/' /etc/selinux/config

echo "===== Disabling swap ====="
swapoff -a
sed -i '/ swap / s/^/#/' /etc/fstab

echo "===== Loading kernel modules ====="
cat <<EOF | tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

modprobe overlay
modprobe br_netfilter

echo "===== Setting sysctl params ====="
cat <<EOF | tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF

sysctl --system

echo "===== Configuring firewall ====="
firewall-cmd --permanent --add-port={6443,2379,2380,10250,10251,10252,10257,10259,179}/tcp
firewall-cmd --permanent --add-port=4789/udp
firewall-cmd --reload
firewall-cmd --list-all

echo "===== Removing Docker/Podman ====="
dnf remove -y docker docker-client docker-client-latest docker-common docker-latest \
docker-latest-logrotate docker-logrotate docker-engine podman runc || true

echo "===== Installing containerd ====="
dnf install -y dnf-plugins-core
dnf config-manager --add-repo https://download.docker.com/linux/rhel/docker-ce.repo
dnf install -y containerd.io

mkdir -p /etc/containerd
containerd config default | tee /etc/containerd/config.toml

echo "===== Configuring containerd cgroup ====="
sed -i 's/SystemdCgroup = false/SystemdCgroup = true/' /etc/containerd/config.toml

systemctl enable --now containerd
systemctl restart containerd

echo "===== Adding Kubernetes repo ====="
cat <<EOF | tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://pkgs.k8s.io/core:/stable:/v1.34/rpm/
enabled=1
gpgcheck=1
gpgkey=https://pkgs.k8s.io/core:/stable:/v1.34/rpm/repodata/repomd.xml.key
EOF

echo "===== Installing Kubernetes ====="
dnf install -y kubelet kubeadm kubectl
systemctl enable --now kubelet

echo "===== Initializing Kubernetes Master ====="
kubeadm init --cri-socket unix:///run/containerd/containerd.sock

echo "===== Configuring kubectl ====="
mkdir -p $HOME/.kube
cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
chown $(id -u):$(id -g) $HOME/.kube/config

echo "===== Installing Calico CNI ====="
kubectl apply -f https://raw.githubusercontent.com/projectcalico/calico/v3.27.0/manifests/calico.yaml

echo "===== DONE ====="
echo "Run: kubectl get nodes"
